== Remove LTE userplane ==
/'
note right of epcagent
	This is part of a longer lifecycle to remove an LTE VNF
end note
'/
epcagent -> epcoam ++ : Remove uplane\n{UUID}\nHTTP REST API
	epcoam -> epccpf ++ : {Internal communication}
/'
ref over epccpf
	Internal: take uplane out of service
end ref
'/
	epccpf --> epcoam -- : {Status}
epcoam --> epcagent -- : {Status}

epcagent --> epcupf : \n\nUPF teardown indication

/'
ref over epcupf
	Userplane VNF stop and removal
	(Mm5, Mm6)
end ref
'/
