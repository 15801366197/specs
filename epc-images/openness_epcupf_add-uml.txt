== Add/Update uplane ==

epcagent -> epcoam ++ : Preconfigure Userplane\n{UUID, APN, ECGI|TAC+CGI}\nHTTP REST API	
	epcoam -> epccpf ++ :  {Internal communication}
	epccpf --> epcoam -- : SUCCESS
epcoam --> epcagent -- : OK

/'
ref over epcupf
	Userplane VNF deployment and start
	(Mm5, Mm6)
end ref
'/

epcagent -> epcupf ++ : \n\nOnboarding userplane component
	epcupf -> epccpf ++ : Register
	epccpf --> epcupf -- : OK
epcupf --> epcagent -- : Ready for service\n(status update)

epcagent -> epcoam ++ : \n\nOptional: Post configuration (Update)
/'
note right of epcagent
	Update UPF NFV status to CPF
	Optional params: IP addresses 
	{sgws1u,sgws5u,sxa,pgws5u,sxb,pridns,secdns}
end note 
'/
epcoam --> epcagent -- : OK
